import requests from bs4 import BeautifulSoup URL = "https://www.espn.com.br/futebol/classificacao/_/liga/BRA.1" HEADERS = { "User-Agent": "Mozilla/5.0" } response = requests.get(URL, headers=HEADERS) soup = BeautifulSoup(response.text, 'html.parser') # Encontrar a tabela principal tabela = soup.find('table') # Se não achou, algo deu errado if not tabela: raise Exception("Tabela não encontrada. Verifique se o site mudou o layout.") # Gerar HTML html_final = f""" Tabela Brasileirão Série A

Tabela do Brasileirão Série A - ESPN

{str(tabela)} """ # Salvar no arquivo with open("index.html", "w", encoding="utf-8") as f: f.write(html_final) print("Arquivo index.html gerado com sucesso!")
